# Conexão WebSocket

Conecte ao endpoint de transporte websocket para gerenciar chamadas de telefonia externa. Este endpoint permite streaming de áudio em tempo real e controle de chamadas para provedores de telefonia externos.

## Visão Geral

O endpoint de Conexão WebSocket fornece o canal de comunicação em tempo real para provedores de telefonia externos:
- Transmitir dados de áudio para e do agente de voz com IA
- Receber mensagens de controle de chamada e atualizações de status
- Gerenciar eventos do ciclo de vida da chamada
- Gerenciar qualidade de áudio e estado da conexão

## Endpoint

O endpoint de transporte WebSocket está disponível em três localizações regionais:

**Estados Unidos:**
```
WebSocket: wss://us-agent.talqover.app/{call_id}/transport?x-talq={token}
```

**Brasil:**
```
WebSocket: wss://br-agent.talqover.app/{call_id}/transport?x-talq={token}
```

**Europa:**
```
WebSocket: wss://eu-agent.talqover.app/{call_id}/transport?x-talq={token}
```

Escolha o endpoint mais próximo da sua localização para melhor performance e menor latência.

## Parâmetros de Conexão

<ParamField path="call_id" type="string" required>
  O ID da chamada retornado do endpoint Fazer Chamada Externa.
</ParamField>

<ParamField query="x-talq" type="string" required>
  Token de autenticação para a conexão websocket. Este token é fornecido na resposta de criação da chamada.
</ParamField>

## Processo de Conexão

1. **Obter ID da Chamada** - Crie uma chamada externa usando o endpoint Fazer Chamada Externa
2. **Extrair Token** - Obtenha o token de autenticação da resposta da chamada
3. **Conectar WebSocket** - Estabeleça conexão com o endpoint de transporte
4. **Gerenciar Streams de Áudio** - Comece a transmitir dados de áudio para a chamada

##### Cliente WebSocket JavaScript

```javascript
const WebSocket = require('ws');

const call_id = '123e4567-e89b-12d3-a456-426614174000';
const token = 'your_websocket_token_here';
// Escolha o endpoint regional apropriado
const usEndpoint = `wss://us-agent.talqover.app/${call_id}/transport?x-talq=${token}`;
const brEndpoint = `wss://br-agent.talqover.app/${call_id}/transport?x-talq=${token}`;
const euEndpoint = `wss://eu-agent.talqover.app/${call_id}/transport?x-talq=${token}`;

// Use o endpoint mais próximo da sua localização
const wsUrl = usEndpoint; // ou brEndpoint, euEndpoint

const ws = new WebSocket(wsUrl);

// Conexão aberta
ws.on('open', () => {
  console.log('WebSocket conectado com sucesso');
  
  // Enviar mensagem inicial de conexão (opcional)
  ws.send(JSON.stringify({
    event: 'clear'
  }));
});

// Mensagem recebida
ws.on('message', (data) => {
  try {
    const message = JSON.parse(data);
    console.log('Mensagem recebida:', message);
    
    switch (message.event) {
      case 'media':
        // Gerenciar áudio recebido do agente
        if (message.media && message.media.payload) {
          handleAgentAudio(message.media.payload);
        }
        break;
      case 'dtmf':
        // Gerenciar tecla DTMF
        if (message.dtmf && message.dtmf.digit) {
          handleDTMF(message.dtmf.digit);
        }
        break;
      default:
        console.log('Tipo de evento desconhecido:', message.event);
    }
  } catch (error) {
    console.error('Erro ao analisar mensagem:', error);
  }
});

// Conexão fechada
ws.on('close', (code, reason) => {
  console.log('WebSocket desconectado:', code, reason);
});

// Erro de conexão
ws.on('error', (error) => {
  console.error('Erro WebSocket:', error);
});

// Gerenciar áudio do agente
function handleAgentAudio(audioData) {
  // Processar os dados de áudio codificados em base64 do agente IA
  // Isso pode envolver reproduzir áudio, gravar ou outro processamento
  console.log('Processando áudio do agente:', audioData.length, 'caracteres (base64)');
}

// Gerenciar tecla DTMF
function handleDTMF(digit) {
  // Processar tecla DTMF do agente
  console.log('Tecla DTMF recebida:', digit);
}
```

##### Node.js WebSocket com Gerenciamento de Áudio

```javascript
const WebSocket = require('ws');
const fs = require('fs');

class ExternalTelephonyClient {
  constructor(call_id, token) {
    this.call_id = call_id;
    this.token = token;
    this.ws = null;
    this.isConnected = false;
  }

  connect() {
    // Escolha o endpoint regional apropriado
    const usEndpoint = `wss://us-agent.talqover.app/${this.call_id}/transport?x-talq=${this.token}`;
    const brEndpoint = `wss://br-agent.talqover.app/${this.call_id}/transport?x-talq=${this.token}`;
    const euEndpoint = `wss://eu-agent.talqover.app/${this.call_id}/transport?x-talq=${this.token}`;

// Use o endpoint mais próximo da sua localização
const wsUrl = usEndpoint; // ou brEndpoint, euEndpoint
    
    this.ws = new WebSocket(wsUrl);
    
    this.ws.on('open', () => {
      console.log('Conectado ao WebSocket do Talqover');
      this.isConnected = true;
      this.sendMessage({
        event: 'clear'
      });
    });

    this.ws.on('message', (data) => {
      this.handleMessage(data);
    });

    this.ws.on('close', () => {
      console.log('Conexão WebSocket fechada');
      this.isConnected = false;
    });

    this.ws.on('error', (error) => {
      console.error('Erro WebSocket:', error);
    });
  }

  sendMessage(message) {
    if (this.isConnected) {
      this.ws.send(JSON.stringify(message));
    }
  }

  sendAudio(audioData) {
    if (this.isConnected) {
      this.ws.send(JSON.stringify({
        event: 'media',
        media: {
          payload: audioData.toString('base64')
        }
      }));
    }
  }

  handleMessage(data) {
    try {
      const message = JSON.parse(data);
      
      switch (message.event) {
        case 'media':
          // Gerenciar áudio recebido do agente
          if (message.media && message.media.payload) {
            this.handleAgentAudio(message.media.payload);
          }
          break;
        case 'dtmf':
          // Gerenciar tecla DTMF
          if (message.dtmf && message.dtmf.digit) {
            this.handleDTMF(message.dtmf.digit);
          }
          break;
        default:
          console.log('Tipo de evento desconhecido:', message.event);
      }
    } catch (error) {
      console.error('Erro ao analisar mensagem:', error);
    }
  }

  handleAgentAudio(audioData) {
    // Gerenciar áudio codificado em base64 recebido do agente IA
    // Aqui é onde você processaria o áudio para seu sistema de telefonia
    console.log('Áudio do agente recebido:', audioData.length, 'caracteres (base64)');
  }

  handleDTMF(digit) {
    // Gerenciar tecla DTMF do agente
    console.log('Tecla DTMF recebida:', digit);
  }

  disconnect() {
    if (this.ws) {
      this.ws.close();
    }
  }
}

// Exemplo de uso
const client = new ExternalTelephonyClient(
  '123e4567-e89b-12d3-a456-426614174000',
  'your_websocket_token_here'
);

client.connect();

// Enviar dados de áudio (exemplo)
setTimeout(() => {
  const audioData = Buffer.from('audio_data_here', 'base64');
  client.sendAudio(audioData);
}, 1000);

// Enviar DTMF (exemplo)
setTimeout(() => {
  client.sendMessage({
    event: 'dtmf',
    dtmf: { digit: '1' }
  });
}, 2000);
```

## Protocolo de Eventos

A conexão WebSocket usa um protocolo padronizado baseado em eventos para comunicação. Todas as mensagens seguem a mesma estrutura com um campo `event` indicando o tipo da mensagem.

### Mensagens de Saída (Cliente → Servidor)

| Evento | Descrição | Carga |
|--------|-----------|-------|
| `media` | Envia dados de áudio para o agente IA | `{ "event": "media", "media": { "payload": "base64_audio" } }` |
| `clear` | Limpa o estado de interrupção | `{ "event": "clear" }` |

### Mensagens de Entrada (Servidor → Cliente)

| Evento | Descrição | Carga |
|--------|-----------|-------|
| `media` | Dados de áudio do agente IA | `{ "event": "media", "media": { "payload": "base64_audio" } }` |
| `dtmf` | Eventos de tecla DTMF | `{ "event": "dtmf", "dtmf": { "digit": "1" } }` |

### Estrutura da Mensagem

Todas as mensagens seguem esta estrutura JSON:
```json
{
  "event": "tipo_evento",
  "campos_adicionais": "valores"
}
```

O campo `event` é sempre obrigatório e determina como a mensagem é processada.

## Formato de Áudio

A conexão WebSocket gerencia dados de áudio no seguinte formato:

- **Codificação**: Dados de áudio codificados em Base64
- **Formato**: PCMU (μ-law) ou PCMA (A-law) a 8kHz
- **Direção**: Streaming bidirecional
- **Taxa de Amostragem**: 8kHz
- **Canais**: Mono (1 canal)

## Estados da Conexão

### Ciclo de Vida da Conexão

1. **Conectando** - Tentativa de conexão WebSocket
2. **Conectado** - Conexão WebSocket estabelecida
3. **Pronto** - Chamada está pronta para começar
4. **Ativo** - Chamada está em andamento
5. **Encerrando** - Chamada está sendo encerrada
6. **Desconectado** - Conexão WebSocket fechada

### Monitoramento de Status

Monitore o estado da conexão para garantir gerenciamento adequado da chamada:

```javascript
ws.on('open', () => {
  console.log('Estado da conexão: Conectado');
});

ws.on('close', () => {
  console.log('Estado da conexão: Desconectado');
});

ws.on('error', () => {
  console.log('Estado da conexão: Erro');
});
```

## Tratamento de Erros

### Erros WebSocket Comuns

| Erro | Descrição | Solução |
|------|-----------|---------|
| `ECONNREFUSED` | Conexão recusada | Verifique se o endpoint está acessível |
| `ETIMEDOUT` | Timeout da conexão | Verifique a conectividade da rede |
| `ECONNRESET` | Conexão resetada | Reconecte com backoff exponencial |
| `Unexpected server response` | Resposta inválida do servidor | Verifique o token de autenticação |

### Estratégia de Reconexão

Implemente uma estratégia robusta de reconexão:

```javascript
class WebSocketManager {
  constructor(url, options = {}) {
    this.url = url;
    this.options = {
      maxRetries: 5,
      retryDelay: 1000,
      ...options
    };
    this.retryCount = 0;
    this.connect();
  }

  connect() {
    try {
      this.ws = new WebSocket(this.url);
      this.setupEventHandlers();
    } catch (error) {
      this.handleConnectionError(error);
    }
  }

  setupEventHandlers() {
    this.ws.on('open', () => {
      console.log('Conectado com sucesso');
      this.retryCount = 0;
    });

    this.ws.on('close', () => {
      console.log('Conexão fechada');
      this.scheduleReconnect();
    });

    this.ws.on('error', (error) => {
      console.error('Erro WebSocket:', error);
      this.handleConnectionError(error);
    });
  }

  scheduleReconnect() {
    if (this.retryCount < this.options.maxRetries) {
      const delay = this.options.retryDelay * Math.pow(2, this.retryCount);
      setTimeout(() => {
        this.retryCount++;
        console.log(`Reconectando... Tentativa ${this.retryCount}`);
        this.connect();
      }, delay);
    }
  }

  handleConnectionError(error) {
    console.error('Erro de conexão:', error);
    this.scheduleReconnect();
  }
}
```

## Melhores Práticas

1. **Conexão Imediata** - Conecte ao WebSocket imediatamente após a criação da chamada
2. **Monitoramento de Heartbeat** - Implemente monitoramento de saúde da conexão
3. **Buffer de Áudio** - Faça buffer dos dados de áudio para lidar com latência da rede
4. **Recuperação de Erros** - Implemente reconexão automática com backoff exponencial
5. **Limpeza de Recursos** - Feche conexões adequadamente e limpe recursos

## Considerações de Segurança

<Info>
  **Autenticação Obrigatória.** Sempre inclua o token `x-talq` na URL WebSocket.
</Info>

<Info>
  **Conexão Segura.** Use WSS (WebSocket Secure) para ambientes de produção.
</Info>

<Info>
  **Validação de Token.** Valide o token de autenticação antes de estabelecer a conexão.
</Info>

## Otimização de Performance

- **Chunking de Áudio** - Envie áudio em tamanhos de chunk apropriados
- **Pool de Conexões** - Reutilize conexões WebSocket quando possível
- **Compressão** - Considere compressão de áudio para otimização de largura de banda
- **Monitoramento de Latência** - Monitore e otimize para streaming de áudio de baixa latência
